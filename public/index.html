Authentication Endpoints

POST /api/auth/register

Register a new user

POST /api/auth/login

Login user

GET /api/auth/me

Get current user profile

PUT /api/auth/updatedetails

Update user details

PUT /api/auth/updatepassword

Update user password

Cart Endpoints

GET /api/auth/cart

Get user's cart

POST /api/auth/cart

Add item to cart

DELETE /api/auth/cart/:productId

Remove item from cart

DELETE /api/auth/cart

Clear cart

Wishlist Endpoints

GET /api/auth/wishlist

Get user's wishlist

POST /api/auth/wishlist

Toggle item in wishlist

DELETE /api/auth/wishlist

Clear wishlist

Product Endpoints

GET /api/products

Get all products

GET /api/products/:id

Get single product

POST /api/products

Create product

PUT /api/products/:id

Update product

DELETE /api/products/:id

Delete product

GET /api/products/categories

Get product categories

Review Endpoints

GET /api/reviews

Get all reviews

GET /api/reviews?productId=ID

Get product reviews

GET /api/reviews/product/:productId/rating

Get product rating

POST /api/reviews

Create review

PUT /api/reviews/:id

Update review

DELETE /api/reviews/:id

Delete review

Order Endpoints

GET /api/orders

Get all orders

GET /api/orders/:id

Get single order

PUT /api/orders/:id

Update order status

DELETE /api/orders/:id

Cancel order
